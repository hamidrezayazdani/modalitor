Usage Example


// Basic initialization
const modal = new Modaling({
    id: 'my-modal',
    animation: 'zoom',
    size: 'md',
    closeOnOverlay: true,
    closeOnEscape: true
});

// Show the modal
modal.show();

// Hide the modal
modal.hide();

// Data attributes usage
<button 
    data-modal-target="my-modal"
    data-modal-animation="zoom"
    data-modal-size="lg"
>
    Open Modal
</button>
